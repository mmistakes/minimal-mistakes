# 기초-반복실행구조

1071번
```cpp
#include <stdio.h>

int main()
{
	int n;
	reload:
		scanf("%d", &n);
		if(n!=0) 
		{
		    printf("%d\n", n);
		    goto reload;
		}
	return 0;
}
```

1072번
```cpp
#include <stdio.h>
int main()
{
	int n, m;
  scanf("%d", &n);
  reget: //레이블은 콜론( : ) 으로 끝난다.
	scanf("%d", &m);
	if (n!=0)
	    {
	        printf("%d\n", m);
	    }
	if(n-- != 0) goto reget; //reget:으로 이동, n의 값 1만큼 감소
	return 0;
}
```
