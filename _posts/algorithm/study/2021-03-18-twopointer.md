    int data;

    bool operator()(const A& a, const A& b) {
        return a.data > b.data;
    }
};

int main() {
    /* μ…μ¶λ ¥ μ†λ„ λ†’μ΄κΈ° */

        cin >> arr[i];

    priority_queue<A, vector<A>, cmp> pq;
    for (int i = 0; i < N; ++i) {
        pq.push({ arr[i], i });
        int pos = pq.top().pos;

    }
    return 0;
}
