    {

            sum += pow(3, x);
            x++;
        }
        else
            break;
    }

    n = n - sum - 1;

    for(int i = 0; i < x; i++)
    {
        answer.append(num[n % 3]);
        n /= 3;
    } 

    reverse(answer.begin(), answer.end());

    return answer;
}
