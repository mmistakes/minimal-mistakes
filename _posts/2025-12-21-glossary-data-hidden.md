(function() {

  const CACHE_KEY = "AgingHealthGlossaryCache";
  const CACHE_TIME_KEY = "AgingHealthGlossaryCacheTime";
  const CACHE_TTL = 24 * 60 * 60 * 1000; // 24 hours

  function isStorageAvailable() {
    try {
      const test = "__test__";
      localStorage.setItem(test, test);
      localStorage.removeItem(test);
      return true;
    } catch (e) {
      return false;
    }
  }

  function loadFromCache() {
    if (!isStorageAvailable()) return null;

    const saved = localStorage.getItem(CACHE_KEY);
    const savedTime = localStorage.getItem(CACHE_TIME_KEY);

    if (!saved || !savedTime) return null;

    const age = Date.now() - parseInt(savedTime, 10);
    if (age > CACHE_TTL) return null;

    return saved;
  }

  function saveToCache(json) {
    if (!isStorageAvailable()) return;
    localStorage.setItem(CACHE_KEY, json);
    localStorage.setItem(CACHE_TIME_KEY, Date.now().toString());
  }

  function applyCache(json) {
    window.AgingHealthGlossaryJSON = json;
    try {
      window.AgingHealthGlossary = JSON.parse(json);
    } catch (e) {
      console.warn("Glossary cache parse error:", e);

/* ============================================================
   AGING HEALTH — GLOSSARY SYNC CHECKER v1
   Validates glossary entries + JSON endpoint consistency
   ============================================================ */

(function() {

  function log(type, msg) {
    console[type]("%c[Glossary Sync] " + msg, "color:#2b6cb0; font-weight:bold;");
  }

  function waitForGlossary(callback) {
    if (window.AgingHealthGlossary) callback(window.AgingHealthGlossary);
    else setTimeout(() => waitForGlossary(callback), 50);
  }

  waitForGlossary(function(glossary) {

    const seen = new Set();
    let errors = 0;

    glossary.forEach(item => {
      const term = item.term || "";
      const def = item.definition || "";
      const cat = item.category || "";
      const icon = item.icon || "";

      if (!term.trim()) {
        log("error", "Missing term in entry: " + JSON.stringify(item));
        errors++;
      }

      if (!def.trim()) {
        log("error", "Missing definition for term: " + term);
        errors++;
      }

      if (!cat.trim()) {
        log("warn", "Missing category for term: " + term);
      }

      if (!icon.trim()) {
        log("warn", "Missing icon for term: " + term);
      }

      const key = term.toLowerCase();
      if (seen.has(key)) {
        log("error", "Duplicate glossary term detected: " + term);
        errors++;
      }
      seen.add(key);
    });

    if (errors === 0) {
      log("log", "Glossary sync OK — no critical issues detected.");
    } else {
      log("error", `Glossary sync completed with ${errors} issue(s).`);
    }

  });

})();
</script>
