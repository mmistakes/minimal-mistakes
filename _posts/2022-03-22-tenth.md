```python
example = [[1,2,3],[4,[5,6]],7,[8,[9]]]
def unlist(line):
    result=[]
    for item in line:
        if type(item) == list:
            result += unlist(item)
        else:
            result += [item]
    return result
print(unlist(example))
```
