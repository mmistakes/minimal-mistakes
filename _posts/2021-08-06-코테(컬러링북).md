void dfs(int i, int j, int m, int n, vector<vector<int>> picture, int color)
{
    int dx[4] = {1, 0, -1, 0};
    int dy[4] = {0, -1, 0, 1};
    check[i][j] = true;
    cnt++;
    for (int c = 0; c < 4; c++)
    {
        int x = i + dx[c];
        int y = j + dy[c];
        if(x>=m || x<0 || y>=n || y<0)
        continue;
        if (!check[x][y] && picture[x][y] == color)
        {
            dfs(x, y, m, n, picture, color);
        }
    }
}
vector<int> solution(int m, int n, vector<vector<int>> picture)
{
    check.assign(m,vector<bool>(n,false));
   
    int number_of_area = 0;
    int max_size_of_one_area = 0;
    cnt=0;
    
//    bool check[100][100];
    //vector<vector<bool>> check(m, vector<bool>(n,0));
//    check.assign(m, vector<bool>(n,0));
    for (int i = 0; i < m; i++)
    {
        for (int j = 0; j < n; j++)
        {
            if (!check[i][j] && picture[i][j] != 0)
            {
                number_of_area++;
                check[i][j]=true;
                dfs(i, j, m, n, picture, picture[i][j]);
                if (max_size_of_one_area < cnt) max_size_of_one_area=cnt;
                cnt=0;
            }
        }
    }
    vector<int> answer(2);
    answer[0] = number_of_area;
    answer[1] = max_size_of_one_area;

    return answer;
}

    vector<vector<int>> picture = {{1, 1, 1, 0}, {1, 2, 2, 0}, {1, 0, 0, 1}, {0, 0, 0, 1}, {0, 0, 0, 3}, {0, 0, 0, 3}};
    vector<int> answer = solution(m, n, picture);
    for (int i : answer)

        cout << i << endl;
